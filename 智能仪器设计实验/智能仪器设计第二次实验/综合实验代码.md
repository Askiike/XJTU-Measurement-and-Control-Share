ORG 0000H

SJMP START

ORG 0030H

 

START:

; LCD初始化

MOV DPTR,#8000H

MOV A,#38H

MOVX @DPTR,A

LCALL DELAY

 

MOV A,#0CH

MOVX @DPTR,A

LCALL DELAY

 

MOV A,#01H

MOVX @DPTR,A

LCALL DELAY

 

MOV A,#80H

MOVX @DPTR,A

LCALL DELAY

 

;=======================

; 初始化显示

;=======================

MOV R3,#00H     ; 默认模式0（与FPGA对应）

MOV R4,#01H     ; 默认 RUN (1=RUN, 0=STOP)

LCALL DISPLAY_ALL

 

; 初始化FPGA控制

MOV A,R3       ; 发送模式0

MOV P1,A

CJNE R4,#01H, INIT_STOP

CLR P1.7       ; RUN

SJMP INIT_DONE

INIT_STOP:

SETB P1.7      ; STOP

INIT_DONE:

 

;=======================

; 按键扫描

;=======================

SCAN_KEY:

MOV DPTR,#0400H

MOV A,#01H

MOVX @DPTR,A

MOVX A,@DPTR

 

; 检查K1

CJNE A,#0FEH, CHECK_K2

LCALL DELAY_DEBOUNCE

MOV DPTR,#0400H

MOV A,#01H

MOVX @DPTR,A

MOVX A,@DPTR

CJNE A,#0FEH, CHECK_K2

 

; 切换模式 0→1→2→0

MOV A,R3

CJNE A,#00H, TO_M1

MOV R3,#01H

LJMP UPDATE_DISP

TO_M1:

CJNE A,#01H, TO_M2

MOV R3,#02H

LJMP UPDATE_DISP

TO_M2:

MOV R3,#00H

 

UPDATE_DISP:

LCALL DISPLAY_ALL

LCALL KEY_UP

LJMP SCAN_KEY

 

; 检查K2

CHECK_K2:

CJNE A,#0FDH, SCAN_KEY

LCALL DELAY_DEBOUNCE

MOV DPTR,#0400H

MOV A,#01H

MOVX @DPTR,A

MOVX A,@DPTR

CJNE A,#0FDH, SCAN_KEY

 

; 切换RUN/STOP

MOV A,R4

XRL A,#01H

MOV R4,A

LCALL DISPLAY_ALL

LCALL KEY_UP

LJMP SCAN_KEY

 

 

;=======================

; 显示 RUN X 或 STOP X

; X显示为 1/2/3 对应模式0/1/2

;=======================

DISPLAY_ALL:

; 第一行：RUN 或 STOP

MOV DPTR,#8000H

MOV A,#80H

MOVX @DPTR,A

LCALL DELAY

 

MOV DPTR,#0A000H

 

; 显示 RUN 或 STOP

MOV A,R4

CJNE A,#01H, SHOW_STOP

 

; 显示 "RUN "

MOV A,#'R'

MOVX @DPTR,A

LCALL DELAY

MOV A,#'U'

MOVX @DPTR,A

LCALL DELAY

MOV A,#'N'

MOVX @DPTR,A

LCALL DELAY

MOV A,#' '

MOVX @DPTR,A

LCALL DELAY

SJMP SHOW_NUMBER

 

SHOW_STOP:

; 显示 "STOP "

MOV A,#'S'

MOVX @DPTR,A

LCALL DELAY

MOV A,#'T'

MOVX @DPTR,A

LCALL DELAY

MOV A,#'O'

MOVX @DPTR,A

LCALL DELAY

MOV A,#'P'

MOVX @DPTR,A

LCALL DELAY

 

SHOW_NUMBER:

; 显示模式数字（将0/1/2转换为'1'/'2'/'3'显示）

MOV A,R3

CJNE A,#00H, DISP_M1

MOV A,#'1'

SJMP DISP_NUM

DISP_M1:

CJNE A,#01H, DISP_M2

MOV A,#'2'

SJMP DISP_NUM

DISP_M2:

MOV A,#'3'

DISP_NUM:

MOVX @DPTR,A

LCALL DELAY

 

; 发送模式到FPGA（直接发送0/1/2）

MOV A,R3

MOV P1,A

 

; 发送RUN/STOP信号

MOV A,R4

CJNE A,#01H, SEND_STOP

CLR P1.7       ; RUN

RET

SEND_STOP:

SETB P1.7      ; STOP

RET

 

 

;=======================

; 消抖延时

;=======================

DELAY_DEBOUNCE:

MOV R5,#400

DL1:

MOV R6,#100

DL2:

DJNZ R6,DL2

DJNZ R5,DL1

RET

 

 

;=======================

; 等待松手

;=======================

KEY_UP:

MOV DPTR,#0400H

MOV A,#01H

MOVX @DPTR,A

MOVX A,@DPTR

CJNE A,#0FFH, KEY_UP

RET

 

 

;=======================

; LCD短延时

;=======================

DELAY:

MOV R5,#20

D3:

MOV R6,#10

D2:

MOV R7,#10

D1:

NOP

DJNZ R7,D1

DJNZ R6,D2

DJNZ R5,D3

RET

 

 

END



 

`timescale 1ns / 1ps

module FPGA_MCU_LCD_KEY(

  clk200Mn, clk200Mp, indicator,

  mcuad7_0, mcua15_8, mcupsen, mcuale, mcuwr, mcurd, mcup3_2,

  lvt245db7_0, lvt245db15_8, lvt245dbdir, lvt245ab15_0, lvt245abdir,

  lcdpsb, lcdrs, lcdrw, lcden,

  keyrow2_0, keycol2_0, testpin

);

 

  input clk200Mn, clk200Mp;

  output reg indicator;

  

  inout [7:0] mcuad7_0;

  input [7:0] mcua15_8;

  input mcupsen, mcuale;

  input mcuwr, mcurd;

  input mcup3_2;

  

  inout [7:0] lvt245db7_0;

  inout [7:0] lvt245db15_8;

  assign lvt245db15_8 = (mcurd) ? 8'b11111111 : 8'bz;

  output lvt245dbdir;

  output [15:0] lvt245ab15_0;

  assign lvt245ab15_0 = 16'hFFFF;

  output lvt245abdir;

  assign lvt245abdir = 1'b0;

  

  wire [7:0] lcddb7_0;

  assign lcddb7_0 = lvt245db7_0;

  output lcdpsb, lcdrs, lcdrw, lcden;

  

  output [2:0] keyrow2_0;

  input [2:0] keycol2_0;

  output reg [7:0] testpin;

  

  // 时钟生成

  wire clk200M;

  wire clk200k;

  FPGA_CLK_200M clk200k_entity(

​    .data_n(clk200Mn),

​    .data_p(clk200Mp),

​    .data_O200M(clk200M),

​    .data_O200k(clk200k)

  );

  

  // MCU与LCD接口

  wire a15, a14, a13;

  assign lvt245db7_0 = (mcurd) ? mcuad7_0 : 8'bz;

  assign a15 = mcua15_8[7];

  assign a14 = mcua15_8[6];

  assign a13 = mcua15_8[5];

  assign lcdpsb = 1'b1;

  assign lcden = !(!a15 || (mcuwr && mcurd));

  assign lcdrw = a14;

  assign lcdrs = a13;

  

  // MCU与矩阵键盘接口

  reg [2:0] keylatch;

  wire [2:0] keyrowcode;

  assign keyrowcode = (!(a15 || mcuwr)) ? mcuad7_0[2:0] : keylatch;

  always @(posedge clk200M)

​    keylatch <= keyrowcode;

  assign keyrow2_0 = keylatch;

  

  // 数据总线仲裁

  wire lcdkeysel;

  wire [7:0] keydata;

  wire [7:0] innerdb7_0;

  assign lcdkeysel = a15;

  assign keydata[2:0] = keycol2_0;

  assign keydata[7:3] = 5'b11111;

  assign innerdb7_0 = (lcdkeysel) ? lcddb7_0 : keydata;

  assign mcuad7_0 = (!mcurd) ? innerdb7_0 : 8'bz;

  assign lvt245dbdir = !mcurd;

  

  // 10kHz 时钟

  reg [15:0] count10;

  reg data_10;

  always @(posedge clk200k) begin

​    count10 <= count10 + 16'd1;

​    if (count10 < 16'd10000)

​      data_10 <= 1'b1;

​    else if (count10 < 16'd20000)

​      data_10 <= 1'b0;

​    else

​      count10 <= 16'd0;

  end

  

  // ================== 按键消抖（逐位独立，互不干扰）==================

  reg [11:0] cnt_k0, cnt_k1, cnt_k2;

  reg k0_stable, k1_stable, k2_stable; // 消抖后的稳定值

  

  // 按键0（keydata[0]）消抖

  always @(posedge clk200k) begin

​    if (keydata[0] == 1'b0) begin

​      cnt_k0 <= cnt_k0 + 1;

​      if (cnt_k0 == 3000) begin

​        k0_stable <= 1'b0;

​        cnt_k0 <= cnt_k0;

​      end

​    end else begin

​      cnt_k0 <= 0;

​      k0_stable <= 1'b1;

​    end

  end

  

  // 按键1（keydata[1]）消抖

  always @(posedge clk200k) begin

​    if (keydata[1] == 1'b0) begin

​      cnt_k1 <= cnt_k1 + 1;

​      if (cnt_k1 == 3000) begin

​        k1_stable <= 1'b0;

​        cnt_k1 <= cnt_k1;

​      end

​    end else begin

​      cnt_k1 <= 0;

​      k1_stable <= 1'b1;

​    end

  end

  

  // 按键2（keydata[2]）消抖

  always @(posedge clk200k) begin

​    if (keydata[2] == 1'b0) begin

​      cnt_k2 <= cnt_k2 + 1;

​      if (cnt_k2 == 3000) begin

​        k2_stable <= 1'b0;

​        cnt_k2 <= cnt_k2;

​      end

​    end else begin

​      cnt_k2 <= 0;

​      k2_stable <= 1'b1;

​    end

  end

  

  // ================== 模式切换 & 启停（边沿检测）==================

  reg [2:0] b = 3'd0;     // 模式 0~2

  reg enable = 1'b1;     // 启停使能，默认运行

  reg k0_last = 1'b1;     // 按键0上一次状态

  reg k1_last = 1'b1;     // 按键1上一次状态

  reg k2_last = 1'b1;     // 按键2上一次状态

  

  always @(posedge data_10) begin

​    // 按键0（keydata[0]）下降沿 → 切换模式

​    if (k0_stable == 1'b0 && k0_last == 1'b1)

​      b <= (b == 3'd2) ? 3'd0 : (b + 1);

​    k0_last <= k0_stable;

​    

​    // 按键1（keydata[1]）下降沿 → 启停翻转

​    if (k1_stable == 1'b0 && k1_last == 1'b1)

​      enable <= ~enable;

​    k1_last <= k1_stable;

​    

​    // 按键2（keydata[2]）下降沿 → 备用（可自行定义功能）

​    // if (k2_stable == 1'b0 && k2_last == 1'b1)

​    //   ... 

​    k2_last <= k2_stable;

  end

  

  // ================== 3种灯效 + 启停控制 ==================

  reg [2:0] a;

  always @(posedge data_10) begin

​    indicator <= 1'b0;

​    if (enable) begin

​      case(b)

​        3'd0: begin // 闪烁模式

​          a <= a - 1'd1;

​          case(a)

​            3'd0: testpin <= 8'b00010001;

​            3'd1: testpin <= 8'b00100010;

​            3'd2: testpin <= 8'b01000100;

​            3'd3: testpin <= 8'b10001000;

​            3'd4: testpin <= 8'b00010001;

​            3'd5: testpin <= 8'b00100010;

​            3'd6: testpin <= 8'b01000100;

​            3'd7: testpin <= 8'b10001000;

​          endcase

​        end

​        3'd1: begin // 正向流水

​          a <= a - 1'd1;

​          case(a)

​            3'd0: testpin <= 8'b00000001;

​            3'd1: testpin <= 8'b00000010;

​            3'd2: testpin <= 8'b00000100;

​            3'd3: testpin <= 8'b00001000;

​            3'd4: testpin <= 8'b00010000;

​            3'd5: testpin <= 8'b00100000;

​            3'd6: testpin <= 8'b01000000;

​            3'd7: testpin <= 8'b10000000;

​          endcase

​        end

​        3'd2: begin // 反向流水

​          a <= a + 1'd1;

​          case(a)

​            3'd0: testpin <= 8'b00000001;

​            3'd1: testpin <= 8'b00000010;

​            3'd2: testpin <= 8'b00000100;

​            3'd3: testpin <= 8'b00001000;

​            3'd4: testpin <= 8'b00010000;

​            3'd5: testpin <= 8'b00100000;

​            3'd6: testpin <= 8'b01000000;

​            3'd7: testpin <= 8'b10000000;

​          endcase

​        end

​        default: ;

​      endcase

​    end else begin

​      testpin <= 8'b00000000; // 停止时全灭

​    end

  end

 

endmodule
